1 /**
   2  * コンテキストメニューを管理するクラス
   3  * @constructor
   4  */
   5 function joContextMenu(thisObj){
   6     this._self = thisObj;
   7 }
   8 joContextMenu.prototype = {
   9 
  10     /**
  11      * コンテキストメニューを開くときにクリックした要素
  12      * @type Node
  13      */
  14     target: null,
  15     
  16     /**
  17      * コンテキストメニューを設定する
  18      * @param {Node} aElement コンテキストメニューを表示させる要素
  19      * @param {Object} context コンテキストメニュー: menupopup(joUI.menupopup)形式
  20      */
  21     set: function(aElement, context){
  22 //        if(!context || typeof context !== 'object') return joCore.logger.error(new Error('Invalid args'));
  23     
  24         this.target = aElement;
  25         this.menu = context;
  26         joEvent.removeEvent(aElement, 'rclick', this.show, false);
  27         joEvent.addEvent(aElement, 'rclick', this.show, false, this);
  28     },
  29     
  30     /**
  31      * 独自のコンテキストメニューを表示しないようにする
  32      * @param {Node} aElement 対象の要素
  33      */
  34     unset: function(aElement){
  35         joEvent.removeEvent(aElement, 'rclick', this.show, false);
  36         joEvent.addEvent(aElement, 'rclick', function(){}, false);
  37     },
  38 
  39     /**
  40      * コンテキストメニューを表示する
  41      * @param {Event} e
  42      */
  43     show: function(e){
  44         //ブラウザ固有のコンテキストメニューが表示されるのを防ぐ
  45         e.preventDefault();
  46         
  47         //本当に期待されている要素のコンテキストメニューを開こうとしているのかどうかを調べる
  48         //もし違っていたら表示しないで終了
  49 //        if(joCore.Node.getUniqueID(joCore.Node.getNodesByPosition(e.clientX, e.clientY)[0]) !== uid) return joCore.logger.log('dont open contextmenu');
  50 
  51         //もしメニューが正しく設定されていなかったら終了
  52         if(!this.menu) return;
  53         
  54         //contextmenu
  55         var ul = joCore.Node.createNode('ul', { 'class': 'menupopup', id: 'contextmenu' }, document.body);
  56         joCore.Node.addStyle(ul, {
  57             position: 'absolute',
  58             top: e.clientY + 'px',
  59             left: e.clientX + 'px',
  60             'z-index': 1000000,
  61         });
  62         ul.appendChild(joUI.menupopup(this.menu, this._self));
  63         joEvent.addEvent(ul, 'click', this.close, false, this);
  64         joEvent.addEvent(ul, 'mousedown', function(){}, false, this);
  65         
  66         //contextmenu が隠れないように補正する
  67         var crect = joCore.Node.getNodeRect(ul); //contextmenu rect
  68         var drect = joCore.Node.getNodeRect($('display')); //display rect
  69         //right
  70         if(crect.right > drect.right)
  71             joCore.Node.addStyle(ul, {
  72                 left: 'auto',
  73                 right: '0'
  74             });
  75         
  76         //bottom
  77         if(crect.bottom > drect.bottom)
  78             joCore.Node.addStyle(ul, {
  79                 top: 'auto',
  80                 bottom: (drect.bottom - e.clientY) + 'px',
  81             });
  82         
  83         //contextmenu を閉じるためのダミーのカバー
  84         var dc = joCore.Node.createNode('div', { id: 'contextmenu-dammy-cover' }, document.body);  //dammy cover
  85         joCore.Node.addStyle(dc, {
  86             position: 'absolute',
  87             top: 0,
  88             left: 0,
  89             width: '100%',
  90             height: '100%',
  91             'z-index': '999999',
  92         });
  93         joEvent.addEvent(dc, 'lclick', this.close, false, this);
  94         joEvent.addEvent(dc, 'rclick', function(e){
  95             this.close(e);
  96         }, false, this);
  97         
  98         //onpopupshowing を呼び出す
  99         if(this.onpopupshowing) this.onpopupshowing(e, this);
 100     },
 101     
 102     /**
 103      * コンテキストメニューを閉じる
 104      * @param {Event} event
 105      */
 106     close: function(event){
 107         joCore.Node.remove($('contextmenu-dammy-cover'));
 108         joCore.Node.remove($('contextmenu'));
 109     },
 110 };
