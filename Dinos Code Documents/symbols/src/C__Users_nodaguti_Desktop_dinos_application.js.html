  70         try{

  74             //スキンファイルをロード
  75             joCore.Node.createNode('style', { 'type': 'text/css', ':context': skinData, 'class': 'application-styles skin-' + id }, uuQuery.tag('head')[0]);
  76             
  77         }catch(e){
  78             joCore.logger.log('An Error has occurred when loading skin of "', this.apps[i].name, '" (', this.apps[i].skin, ')');

 125     /**
 126      * アプリケーションを起動する
 127      * @param {String} appID
 128      * @param {All} param アプリケーションに渡すパラメータ
 129      */
 130     launch: function(appID, param, isStartup){
 131         var appInfo = this.getAppInfoByID(appID);
 132         
 133         if(!appInfo) return;
 134         
 135         var appFile = joCore.IO.Dinos.read(appInfo.app);
 136         
 137         if(!appFile) return;
 138         
 139         if(!joApps[appID]){
 140             try{
 141                 joApps[appID] = {};    
 142             
 143                 //アプリケーションコードをロード
 144                 eval(
 145                     
 146                     '(function(){' +
 147                     
 148                         'var _id_ = "' + appID + '";' + 
 149                         'var _self_;' + 
 150                 
 151                         appFile +
 152                         
 153                         'joApps[_id_] = _self_;' + 
 154                     
 155                     '})();'
 156                 );
 157                 
 158                 if(!joApps[appID]) throw new Error('Invalid type of application.');
 159                 
 160             }catch(e){
 161                 return joCore.logger.error('An Error has occurred when loading "', appInfo.name, '" (', appInfo.app, ')', e);
 162             }
 163         }
 164         
 165         try{
 166         
 167             if(isStartup)
 168                 joApps[appID].onload();
 169             else
 170                 joApps[appID].start(param);
 171                 
 172         }catch(e){
 173             return joCore.logger.error('An Error has occurred when starting "', appInfo.name, '" (', appInfo.app, ')', e);
 174         }
 175     },

 223     /**

 225      *
 226      * @param {String} [appID] 指定したアプリケーションのウィンドウに限定する
 227      * @return {Node} ウィンドウのノード 見つからないときは null を返す
 228      */
 229     getFrontWindow: function(appID){
 230         if(!appID) appID = '';
 231 
 232         var _active = uuQuery('.window.active');
 233         
 234         if(_active.length && joCore.Node.hasClass(_active[0], appID)) return _active[0];
 235 
 236         return (function(){
 237             //[active=true]なwindowがない時はより厳密に判定してみる
 238             var windows = appID ? $C(appID, $('windows')) : $C('window', $('windows'));
 239             var max = 0, frontWindow = null;
 240 
 241             if(!windows) return null;
 242 
 243             for(var i=0,l=windows.length;i<l;i++){
 244                 if(max < windows[i].style.zIndex){
 245                     max = windows[i].style.zIndex;
 246                     frontWindow = windows[i];
 247                 }
 248             }
 249 
 250             return frontWindow;
 251         })();
 252     },
