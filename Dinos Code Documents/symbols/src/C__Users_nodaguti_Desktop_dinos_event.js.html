1 /**
   2  * イベントを取り扱うクラス
   3  * @class
   4  */
   5 var joEvent = {
   6 
   7     /**
   8      * ノードに結び付けられているイベントを管理するマップ
   9      */
  10     _eventMap: {},
  11 
  12     /**
  13      * 起動時に一度だけ実行される
  14      * @private
  15      */
  16     init: function(){
  17         window.addEventListener('mousemove', function(e){
  18             if(joEvent._moveTarget) joEvent._onDrag(e);
  19             else if(joEvent._resizeTarget) joEvent._onResize(e);
  20         }, false);
  21     },

  23     /**
  24      * イベントを登録する
  25      *
  26      * @param {Node} aElement 対象の要素
  27      * @param {String} aType イベントのタイプ <br> 空白で区切ることで複数イベントをリッスン可能 <br> 特殊イベント: (r|l|middle)click, (r|l|middle)mouse(down|up|move), key**(実装予定)
  28      * @param {Function} aCallback コールバック関数 第一引数:eventオブジェクト, 第二引数: triggerdData（fireEventでデータが渡された場合）
  29      * @param {Boolean} [isUseCapture=false] キャプチャフェーズにするかどうか
  30      * @param {Object} [aThisObj=Event] コールバック関数内でthisが指すオブジェクト
  31      * @param {Boolean} [notStopPropagation=false] 自動でstopPropagationするかどうか
  32      */
  33     addEvent: function(aElement, aType, aCallback, isUseCapture, aThisObj, notStopPropagation){
  34         var types = aType.split(' ');
  35         var uid = joCore.Node.getUniqueID(aElement);    
  36     
  37         types.forEach(function(type){
  38             var rtype = this._getRegularType(type);
  39             
  40             //イベントマップに登録する
  41             var eid = joCore.utils.getUniqueID();  //event id
  42             this._eventMap[eid] = {
  43                 element: uid,
  44                 type: type,
  45                 self: aThisObj,
  46                 callback: aCallback,
  47                 propagation: notStopPropagation
  48             };
  49             
  50             this._runClosure[eid] = (function(eid){
  51                 return function(event){
  52                     joEvent._eventFired(event, eid);
  53                 };
  54             })(eid);
  55             
  56             aElement.addEventListener(rtype, this._runClosure[eid], isUseCapture===undefined ? false : isUseCapture);
  57         }, this);
  58     },
  59     
  60     /**
  61      * 独自イベント名をDOM Eventsのイベント名に変換する
  62      * @param {String} type
  63      * @return {String}
  64      */
  65     _getRegularType: function(type){
  66         switch(type){
  67             case 'lclick':
  68             case 'middleclick':
  69                 return 'click';
  70     
  71             case 'rclick':
  72                 return 'contextmenu';
  73                 
  74             case 'rmousedown':
  75             case 'lmousedown':
  76             case 'middlemousedown':
  77                 return 'mousedown';
  78                 
  79             case 'rmouseup':
  80             case 'lmouseup':
  81             case 'middlemouseup':
  82                 return 'mouseup';
  83             
  84             case 'rmousemove':
  85             case 'lmousemove':
  86             case 'middlemove':
  87                 return 'mousemove';
  88                 
  89             case 'dblclick-mod':
  90                 return 'mouseup';
  91             
  92             default:
  93                 return type;
  94         }
  95     },
  96     
  97     /**
  98      * イベントを解除する
  99      * @param {Node} aElement 対象の要素
 100      * @param {String} aType イベントタイプ
 101      * @param {Function} aCallback コールバック関数
 102      * @param {Boolean} isUseCapture キャプチャーフェーズするかどうか
 103      */
 104     removeEvent: function(aElement, aType, aCallback, isUseCapture){
 105         var uid = joCore.Node.getUniqueID(aElement);
 106         var rtype = this._getRegularType(aType);
 107     
 108         for(var i in this._eventMap){
 109             var map = this._eventMap[i];
 110             
 111             if(map.element === uid &&
 112                map.type === aType &&
 113                map.callback === aCallback){
 114                  aElement.removeEventListener(rtype, this._runClosure[i], isUseCapture);
 115                  delete this._eventMap[i];
 116             }
 117         }
 118     },

 120     _runClosure: {},
 121 
 122     /**
 123      * イベントが発火したときに、コールバックを呼び出すための関数
 124      * @param {Event} event
 125      * @param {String} eid Event ID
 126      */
 127     _eventFired: function(event, eid){    
 128         var map = joEvent._eventMap[eid];
 129     
 130         //特殊イベントの判定
 131         switch(map.type){
 132             case 'rmousedown':
 133             case 'rmousemove':
 134             case 'rmouseup':
 135                 if(event.button !== 2) return;
 136                 break;
 137         
 138             case 'lclick':
 139             case 'lmousedown':
 140             case 'lmousemove':
 141             case 'lmouseup':
 142                 if(event.button !== 0) return;
 143                 break;
 144 
 145             case 'middleclick':
 146             case 'middlemousedown':
 147             case 'middlemousemove':
 148             case 'middlemouseup':
 149                 if(event.button !== 1) return;
 150                 break;
 151                 
 152             case 'dblclick-mod':    
 153                 if(this._dblc_phase !== this._dblc_PHASE_ONECLICK ||
 154                    ((new Date()).getTime() - this._dblc_TIME) > this._dblc_CLICK_LIMIT ||
 155                    this._dblc_last_event_id !== eid){
 156                         this._dblc_phase = this._dblc_PHASE_ONECLICK;
 157                         this._dblc_last_event_id = eid;
 158                         this._dblc_TIME = (new Date()).getTime();
 159                         return;
 160                 }
 161                 
 162                 this._dblc_phase = this._dblc_PHASE_NONE;
 163                 this._dblc_TIME = (new Date()).getTime();
 164                 break;
 165         }
 166     
 167         //自動stopPropagation
 168         if(!map.propagation) event.stopPropagation();
 169         
 170         //callbackを呼び出す
 171         map.callback.call(map.self || event.target, event, event._data);
 172     },
 173     _dblc_phase: 0,
 174     _dblc_last_event_id: null,
 175     _dblc_PHASE_NONE: 0,
 176     _dblc_PHASE_ONECLICK: 1,
 177     _dblc_TIME: 0,
 178     _dblc_CLICK_LIMIT: 250,
 179 
 180     /**
 181      * Mutation Event を発火させる
 182      *
 183      * @param {String} aName イベントの名前
 184      * @param {Node} aTarget イベントを発生させる要素
 185      * @param {Object} data イベントcallbackに渡すデータ
 186      */
 187     fireEvent: function(aName, aTarget, data){
 188         var evt = document.createEvent('MutationEvent');
 189         evt.initMutationEvent(aName, true, false, aTarget, null, null, null, null);
 190         
 191         if(data) evt._data = data;
 192         aTarget.dispatchEvent(evt);
 193     },
 194     
 195     stopPropagation: function(aElement, eventType){
 196         this.addEvent(aElement, eventType, function(){}, false);

 200     /**
 201      * 要素をドラッグ可能にする
 202      *
 203      * @param {Node} eventTarget イベントを受け付けるノード
 204      * @param {Node} [moveTarget=] 実際に移動させるノード <br>省略時は eventTarget と同じになる
 205      * @param {Node} aParentNode position: absolute の基準となるノード
 206      */
 207     setDraggable: function(eventTarget, moveTarget, aParentNode){
 208         var eventTarget = eventTarget;
 209         var moveTarget = moveTarget || eventTarget;
 210         var parent = aParentNode;
 211         joEvent.addEvent(eventTarget, 'lmousedown', function(e){ joEvent._onDragStart(e, eventTarget, moveTarget, parent); }, false);
 212     },

 214     _onDragStart: function(event, eventTarget, moveTarget, parent){
 215         var self = joEvent;
 216 
 217         self._moveTarget = moveTarget;
 218         self._eventTarget = eventTarget;
 219         
 220         event.stopPropagation();
 221         event.preventDefault();
 222         
 223         joEvent.addEvent(eventTarget, 'lmousemove', self._onDrag, false);
 224         joEvent.addEvent(eventTarget, 'lmouseup', self._onDragEnd, false);
 225 
 226         var moveTargetRect = joCore.Node.getNodeRect(moveTarget, parent);
 227         self.offsetX = moveTargetRect.left - event.clientX;
 228         self.offsetY = moveTargetRect.top - event.clientY;
 229         
 230         self.fireEvent('joDragStart', moveTarget);
 231     },
 232 
 233     _onDrag: function(event){
 234         var self = joEvent;
 235 
 236         event.stopPropagation();
 237         event.preventDefault();
 238 
 239         joCore.Node.addStyle(self._moveTarget, {
 240             'left': event.clientX + self.offsetX + 'px',
 241             'top': event.clientY + self.offsetY + 'px'
 242         });
 243         
 244         try{
 245             self._moveTarget.owner.x = event.clientX + self.offsetX;
 246             self._moveTarget.owner.y = event.clientY + self.offsetY;
 247         }catch(e){}
 248         
 249         joEvent.fireEvent('joDrag', joEvent._moveTarget);
 250     },
 251 
 252     _onDragEnd: function(event){
 253         var target = joEvent._moveTarget;
 254         joEvent._moveTarget = null;
 255 
 256         event.stopPropagation();
 257         event.preventDefault();
 258 
 259         joEvent.removeEvent(joEvent._eventTarget, 'lmousemove', joEvent._onDrag, false);
 260         joEvent.removeEvent(joEvent._eventTarget, 'lmouseup', joEvent._onDragEnd, false);
 261         
 262         joEvent.fireEvent('joDragEnd', target);
 263     },

 266     /**
 267      * 要素をリサイズ可能にする
 268      *
 269      * @param {Node} eventTarget イベントを受け付けるノード
 270      * @param {Node} [resizeTarget] 実際にリサイズさせるノード <br>省略時は eventTarget と同じになる
 271      */
 272     setResizable: function(eventTarget, resizeTarget){
 273         var eventTarget = eventTarget;
 274         var resizeTarget = resizeTarget || eventTarget;
 275         joEvent.addEvent(eventTarget, 'lmousedown', function(e){ joEvent._onResizeStart(e, eventTarget, resizeTarget); }, false);
 276     },
 277 
 278     _onResizeStart: function(event, eventTarget, resizeTarget){
 279         var self = joEvent;
 280 
 281         self._resizeTarget = resizeTarget;
 282         self._eventTarget = eventTarget;
 283 
 284         event.stopPropagation();
 285         event.preventDefault();
 286 
 287         var bounds = self._resizeTarget.getBoundingClientRect();
 288         self.left = bounds.left;
 289         self.top = bounds.top;
 290         self.offsetX = bounds.right - event.clientX;
 291         self.offsetY = bounds.bottom - event.clientY;
 292 
 293         joEvent.addEvent(eventTarget, 'lmousemove', self._onResize, false);
 294         joEvent.addEvent(eventTarget, 'lmouseup', self._onResizeEnd, false);
 295         
 296         self.fireEvent('joResizeStart', resizeTarget);
 297     },
 298 
 299     _onResize: function(event){
 300         var self = joEvent;
 301 
 302         event.stopPropagation();
 303         event.preventDefault();
 304 
 305         joCore.Node.addStyle(self._resizeTarget, {
 306             width: (event.clientX + self.offsetX - self.left) + 'px',
 307             height: (event.clientY + self.offsetY - self.top) + 'px'
 308         });
 309         
 310         try{
 311             self._resizeTarget.owner.width = (event.clientX + self.offsetX - self.left);
 312             self._resizeTarget.owner.height = (event.clientY + self.offsetY - self.top);
 313         }catch(e){}
 314         
 315         joEvent.fireEvent('joResize', joEvent._resizeTarget);
 316     },
 317 
 318     _onResizeEnd: function(event){
 319         var target = joEvent._resizeTarget;
 320         joEvent._resizeTarget = null;
 321 
 322         event.stopPropagation();
 323         event.preventDefault();
 324 
 325         joEvent.removeEvent(joEvent._eventTarget, 'lmousemove', joEvent._onResize, false);
 326         joEvent.removeEvent(joEvent._eventTarget, 'lmouseup', joEvent._onResizeEnd, false);
 327         
 328         joEvent.fireEvent('joResizeEnd', target);
 329     },
