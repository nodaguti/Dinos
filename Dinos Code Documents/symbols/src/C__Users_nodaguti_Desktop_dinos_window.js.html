 110     /**
 111      * ウィンドウの位置を、ほかのウィンドウを元にして自動で決定する
 112      */
 113     _setRelatedPosition: function(){
 114         if(typeof this.param.related === 'object' || (!this.param.x && !this.param.y) ){
 115             if(this.param.related){
 116                 var node = this.param.related.display ? this.param.related.display.content : this.param.related;
 117             }else{
 118                 var node = joAppManager.getFrontWindow();
 119             }
 120             
 121             if(!node) return;
 122             
 123             var rect = joCore.Node.getNodeRect(node);
 124             this.param.x = rect.left + 10;
 125             this.param.y = rect.top + 10;
 126         }

 129     /**
 130      * ウィンドウの大きさを内容に応じて適切な大きさに変更することを試みる
 131      */
 132     setAppropriateSize: function(){
 133     
 134         //測定用要素の作成
 135         var div = joCore.Node.createNode('div', {
 136             'class': 'jowindow-size-judgement',
 137             ':child': this.display.content.innerHTML,
 138         }, document.body);
 139         
 140         //測定用CSSの追加
 141         joCore.Node.addStyle(div, {
 142             'visibility': 'hidden',
 143             'position': 'absolute',
 144             'top': 0,  //joDisplay.displayHeight + 99 + 'px',
 145             'left': 0, //joDisplay.displayWidth + 99 + 'px',
 146             'width': 'auto',
 147             'height': 'auto',
 148             'line-height': '1.8',
 149         });
 150         
 151         uuQuery('input, button', div).forEach(function(form){
 152             joCore.Node.addStyle(form, { display: 'block' });
 153         });
 154         
 155         
 156         //大きさ測定
 157         var rect = joCore.Node.getNodeRect(div);    
 158         joCore.Node.remove(div);
 159         
 160         //補正
 161         var width = rect.width + 50;
 162         var height = rect.height + 20;
 163         
 164         if(width > joDisplay.displayWidth) width = joDisplay.displayWidth;
 165         if(width < 200) width = 200;
 166         
 167         if(height > joDisplay.displayheight) height = joDisplay.displayheight;
 168         if(height < 100) height = 100;
 169         
 170         //実際にウィンドウのサイズを変更する
 171         this.resizeTo(width, height);
 172     },

 348 /**
 349  * Create a dialog
 350  * @constructor
 351  * @extends joWindow
 352  * @param {String} param.message ダイアログに表示するメッセージ
 353  * @param {Function} [param.callback] OKボタンが押されたときに呼び出される関数
 354  * @note param.width / param.height も省略可能 <br>
 355  *       param.x / param.y がともに未指定の場合は画面中央に表示する
 356  */
 357 function joDialog(param){
 358     this.param = param;
 359     this.display = {};
 360     
 361     this.init();
 362     this.init_self();
 363     
 364     //大きさが設定されていない場合は内容に合わせたサイズにする
 365     if(!this.param.width || !this.param.height) this.setAppropriateSize();
 366     
 367     //位置が設定されていない場合は中央に表示する
 368     if(!this.param.x && !this.param.y) this.moveTo( ( joDisplay.displayWidth  - this.width  )/2,
 369                                                     ( joDisplay.displayHeight - this.height )/2  );
 370 }
